   699→async function generateTitles() {
   700→  const propType = document.getElementById('propType').value;
   701→  const purpose  = document.getElementById('purpose').value;
   702→  const city     = document.getElementById('city').value;
   703→  const area     = document.getElementById('area').value;
   704→  const space    = document.getElementById('space').value;
   705→  const rooms    = document.getElementById('rooms').value;
   706→  const feature  = document.getElementById('feature').value;
   707→  const price    = document.getElementById('price').value;
   708→
   709→  const errorMsg = document.getElementById('errorMsg');
   710→  if (!propType || !purpose || !city) {
   711→    errorMsg.classList.add('show');
   712→    return;
   713→  }
   714→  errorMsg.classList.remove('show');
   715→
   716→  // Build prompt
   717→  const extras = [
   718→    space   ? `المساحة: ${space} م²` : '',
   719→    rooms   ? `عدد الغرف: ${rooms}` : '',
   720→    feature ? `ميزة بارزة: ${feature}` : '',
   721→    price   ? `السعر: ${price}` : '',
   722→    area    ? `الحي/المنطقة: ${area}` : '',
   723→  ].filter(Boolean).join('، ');
   724→
   725→  const prompt = `أنت خبير في التسويق العقاري للسوق العربي والخليجي.
   726→
   727→بيانات العقار:
   728→- النوع: ${propType}
   729→- الغرض: ${purpose}
   730→- المدينة: ${city}
   731→${extras ? `- تفاصيل إضافية: ${extras}` : ''}
   732→
   733→المطلوب: اكتب بالضبط 4 عناوين تسويقية جذابة وقوية لهذا العقار، كل عنوان مخصص لمنصة مختلفة.
   734→
   735→أجب فقط بهذا الـ JSON بدون أي كلام إضافي أو backticks:
   736→{
   737→  "titles": [
   738→    {"platform": "واتساب", "title": "..."},
   739→    {"platform": "إنستغرام", "title": "..."},
   740→    {"platform": "تويتر / X", "title": "..."},
   741→    {"platform": "لينكدإن", "title": "..."}
   742→  ]
   743→}
   744→
   745→قواعد كل عنوان:
   746→- واتساب: تفصيلي وودود، يذكر المميزات والموقع، ينتهي بدعوة للتواصل (لا يتجاوز 3 أسطر)
   747→- إنستغرام: جذاب ومشاعري، يستخدم نقاط وإيموجي مناسبة، قصير وقوي
